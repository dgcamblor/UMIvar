# Notas

You can use `!!sym()` to convert a string to a variable name. 

```
variable <- "relapse"

relapsing_n <- data_wider %>% filter(!!sym(variable) == 1) %>% nrow()
non_relapsing_n <- data_wider %>% filter(!!sym(variable) == 0) %>% nrow()

data_longer %>% 
  group_by(GENE_SYMBOL, !!sym(variable)) %>% 
  summarise(
    "count" = n()
  ) %>% 
  mutate(
    "rel_freq" = count / ifelse(!!sym(variable) == 1, relapsing_n, non_relapsing_n)
  ) %>% 
  filter(GENE_SYMBOL %in% top_mutated_genes$GENE_SYMBOL[1:10]) %>%
  ggplot(aes(x = GENE_SYMBOL, y = rel_freq, fill = factor(!!sym(variable)))) +
  geom_bar(stat = "identity", position = "dodge") +
  coord_flip() +
  labs(
    x = "Gene",
    y = "Frequency",
    title = "Frequency of mutations in relapsing and non-relapsing patients"
  ) +
  scale_fill_discrete(name = "Relapse")
```
